Create Databases

Syntax CREATE DATABASE databasename;

Drop Databases

Syntax DROP DATABASE databasename;

Create Table

Syntax CREATE TABLE table_name ( column1 datatype, column2 datatype, column3 datatype, .... );

Drop Databases

Syntax DROP TABLE table_name;

Select

Syntax SELECT column1, column2, ... FROM table_name;

Where

Syntax SELECT column1, column2, ... FROM table_name WHERE condition;

And,Or$Not

Syntax- And SELECT column1, column2, ... FROM table_name WHERE condition1 AND condition2 AND condition3 ...; Or SELECT column1, column2, ... FROM table_name WHERE condition1 OR condition2 OR condition3 ...; Not SELECT column1, column2, ... FROM table_name WHERE NOT condition;

Insert into

Syntax INSERT INTO table_name (column1, column2, column3, ...) VALUES (value1, value2, value3, ...);

AGGREGATE FUNCTIONS

Syntax CREATE DATABASE databasename;

GROUP BY, HAVING, ORDER BY

yntax-SELECT column_name(s) FROM table_name WHERE condition GROUP BY column_name(s) ORDER BY column_name(s); ,Having ELECT column_name(s) FROM table_name WHERE condition GROUP BY column_name(s) HAVING condition ORDER BY column_name(s);,Order by SELECT column1, column2, ... FROM table_name ORDER BY column1, column2, ... ASC|DESC;

NULL

yntax SELECT column_names FROM table_name WHERE column_name IS NULL;

IN & BETWEEN OPERATORS

Syntax-In SELECT column_name(s) FROM table_name WHERE column_name IN (value1, value2, ...); or: SELECT column_name(s) FROM table_name WHERE column_name IN (SELECT STATEMENT); or Between SELECT column_name(s) FROM table_name WHERE column_name BETWEEN value1 AND value2;

UPDATE & DELETE

Syntax-UpdateUPDATE table_name SET column1 = value1, column2 = value2, ... WHERE condition; or DeleteDELETE FROM table_name WHERE condition;
